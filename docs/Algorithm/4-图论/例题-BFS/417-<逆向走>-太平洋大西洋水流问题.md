题目：[417. 太平洋大西洋水流问题](https://leetcode.cn/problems/pacific-atlantic-water-flow/)

有一个 `m × n` 的矩形岛屿，与 **太平洋** 和 **大西洋** 相邻。 **“太平洋”** 处于大陆的左边界和上边界，而 **“大西洋”** 处于大陆的右边界和下边界。

这个岛被分割成一个由若干方形单元格组成的网格。给定一个 `m x n` 的整数矩阵 `heights` ， `heights[r][c]` 表示坐标 `(r, c)` 上单元格 **高于海平面的高度** 。

岛上雨水较多，如果相邻单元格的高度 **小于或等于** 当前单元格的高度，雨水可以直接向北、南、东、西流向相邻单元格。水可以从海洋附近的任何单元格流入海洋。

返回网格坐标 `result` 的 **2D 列表** ，其中 `result[i] = [ri, ci]` 表示雨水从单元格 `(ri, ci)` 流动 **既可流向太平洋也可流向大西洋** 。

**示例 1：**

![img](https://assets.leetcode.com/uploads/2021/06/08/waterflow-grid.jpg)

```
输入: heights = [[1,2,2,3,5],[3,2,3,4,4],[2,4,5,3,1],[6,7,1,4,5],[5,1,1,2,4]]
输出: [[0,4],[1,3],[1,4],[2,2],[3,0],[3,1],[4,0]]
```

**示例 2：**

```
输入: heights = [[2,1],[1,2]]
输出: [[0,0],[0,1],[1,0],[1,1]]
```

提示：

m == heights.length
n == heights[r].length
1 <= m, n <= 200
`0 <= heights[r][c] <= 105`



```c++
class Solution {
private:
    int rows;
    int cols;
    vector<vector<int>> grid;
    pair<int, int> directions[4] = {{0, 1}, {1, 0}, {-1, 0}, {0, -1}};

    void bfs(queue<pair<int, int>> &que, bool vis[201][201]) {
        while (!que.empty()) {
            int cur_r = que.front().first;
            int cur_c = que.front().second;
            que.pop();
            for (auto &[dr, dc] : directions) {
                int next_r = cur_r + dr;
                int next_c = cur_c + dc;
                if (next_r < 0 || next_r >= rows || next_c < 0 || next_c >= cols) continue;  // 边界外
                if (vis[next_r][next_c]) continue;                                           // 不重复走
                if (grid[next_r][next_c] < grid[cur_r][cur_c]) continue;                     // 往高处走
                que.push({next_r, next_c});
                vis[next_r][next_c] = true;
            }
        }
    }

    void print_vis(bool vis[201][201]) {
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                cout << vis[r][c] << " ";
            }
            cout << endl;
        }
    }

public:
    vector<vector<int>> pacificAtlantic(vector<vector<int>> &heights) {
        // 从边往中间走
        rows = heights.size();
        cols = heights[0].size();
        grid = heights;

        // 流到太平洋
        queue<pair<int, int>> que_pac;
        bool vis_pac[201][201];
        memset(vis_pac, false, sizeof(bool) * 201 * 201);

        for (int c = 0; c < cols; c++) {
            que_pac.push({0, c});
            vis_pac[0][c] = true;
        }
        for (int r = 1; r < rows; r++) {
            que_pac.push({r, 0});
            vis_pac[r][0] = true;
        }
        bfs(que_pac, vis_pac);
        // print_vis(vis_pac);

        // 流到大西洋
        queue<pair<int, int>> que_atl;
        bool vis_atl[201][201];
        memset(vis_atl, false, sizeof(bool) * 201 * 201);

        for (int c = 0; c < cols; c++) {
            que_atl.push({rows - 1, c});
            vis_atl[rows - 1][c] = true;
        }
        for (int r = 0; r < rows - 1; r++) {
            que_atl.push({r, cols - 1});
            vis_atl[r][cols - 1] = true;
        }
        bfs(que_atl, vis_atl);
        // print_vis(vis_atl);

        vector<vector<int>> ans;
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                if (vis_pac[r][c] && vis_atl[r][c]) {
                    ans.push_back({r, c});
                }
            }
        }

        return ans;
    }
};
```

```python
class Solution:
    def main(self, heights):
        rows = len(heights)
        cols = len(heights[0])
        queue_t = []
        queue_d = []
        for r in range(rows):
            queue_t.append((r, 0))
            queue_d.append((r, cols-1))
        for c in range(cols):
            queue_t.append((0, c))
            queue_d.append((rows-1, c))
        def bfs(queue):
            visited = set()
            while queue:
                r, c = queue.pop(0)
                visited.add((r,c))
                for dr, dc in ((1,0),(0,1),(-1,0),(0,-1)):
                    next_r = r + dr
                    next_c = c+dc
                    if 0<=next_r<=rows-1 and 0<=next_c<=cols-1:
                        if heights[next_r][next_c] < heights[r][c]:
                            continue
                        if (next_r, next_c) in visited:
                            continue
                        queue.append((next_r, next_c))
            return visited
        res_t = bfs(queue_t)
        res_d = bfs(queue_d)
        return [list(r) for r in res_t & res_d]
```

```python
class Solution:
    def main(self, heights):
        rows = len(heights)
        cols = len(heights[0])
        res_t = set()
        res_d = set()

        # 从边界遍历
        def dfs(r, c, res):
            res.add((r, c))

            for dr, dc in ((0, 1), (1, 0), (-1, 0), (0, -1)):
                next_r = r + dr
                next_c = c + dc
                if 0 <= next_r <= rows - 1 and 0 <= next_c <= cols - 1:
                    if heights[next_r][next_c] < heights[r][c]:
                        continue
                    if (next_r, next_c) in res:
                        continue
                    dfs(next_r, next_c, res)

        # 太平洋 左边界
        for r in range(rows):
            dfs(r, 0, res_t)
        # 太平洋 上边界
        for c in range(cols):
            dfs(0, c, res_t)
        # 大西洋 右边界
        for r in range(rows):
            dfs(r, cols - 1, res_d)
        # 大西洋 下边界
        for c in range(cols):
            dfs(rows - 1, c, res_d)

        return [list(r) for r in res_t & res_d]
```
